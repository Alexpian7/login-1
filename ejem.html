function calcularNomina() { // Obtener los valores de los campos de entrada const fechaInicio = new Date(document.getElementById('fechaInicio').value); const fechaFinal = new Date(document.getElementById('fechaFinal').value); const auxilioTransporte = parseFloat(document.getElementById('auxilioTransporte').value); const diasSemestre = parseInt(document.getElementById('diastrabajadossemestre').value); const salarioMensual = parseFloat(document.getElementById('salarioMensual').value); // Verificar si las fechas son válidas if (isNaN(fechaInicio.getTime()) || isNaN(fechaFinal.getTime())) { document.getElementById('salarioTotalResult').innerText = 'Por favor, introduce fechas válidas.'; return; } // Calcular los días trabajados const diasTrabajados = (fechaFinal - fechaInicio) / (1000 * 60 * 60 * 24); // Calcular Cesantías const cesantias = ((salarioMensual + auxilioTransporte) * diasTrabajados) / 360; // Calcular Intereses sobre cesantías const interesesCesantias = (cesantias * diasTrabajados * 0.12) / 360; // Calcular Prima de servicios const primaServicios = (salarioMensual * diasSemestre) / 360; // Calcular Vacaciones const vacaciones = (salarioMensual * diasTrabajados) / 720; // Calcular Salario Total const salarioTotal = cesantias + interesesCesantias + primaServicios + vacaciones; // Mostrar los resultados en los elementos HTML document.getElementById("salarioTotalResult").innerText = "Salario Total: $" + salarioTotal.toFixed(2); // Crear un nuevo documento PDF }
